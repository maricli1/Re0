<script>
    // --- Page Transition Logic ---
    document.addEventListener('DOMContentLoaded', () => {
        const startScreen = document.getElementById('start-screen');
        const startBtn = document.getElementById('start-btn');
        const characterSheet = document.getElementById('character-sheet');

        startBtn.addEventListener('click', () => {
            // Hide the start screen and show the character sheet
            startScreen.classList.add('hidden');
            characterSheet.classList.remove('hidden');
        });
    });

    // --- Form Submission Logic ---
    document.getElementById('confirm-btn').addEventListener('click', async () => {
        const getValue = (id, defaultValue, isTextarea = false) => {
            const element = document.getElementById(id);

            // --- MODIFIED SECTION START ---
            if (element.type === 'number') {
                let value = parseInt(element.value, 10);
                const min = parseInt(element.min, 10);
                const max = parseInt(element.max, 10);

                // If input is not a number or empty, use default value
                if (isNaN(value)) {
                    return defaultValue;
                }

                // Clamp the value within the min/max range
                if (value < min) {
                    value = min;
                }
                if (value > max) {
                    value = max;
                }
                return value;
            }
            // --- MODIFIED SECTION END ---

            const trimmedValue = element.value.trim();
            if (trimmedValue === '') return defaultValue;
            if (isTextarea && trimmedValue !== defaultValue) {
                 if (id === 'skills') {
                    const skills = {};
                    trimmedValue.split('\n').forEach(line => {
                        const parts = line.split(/[:：]/);
                        if (parts.length === 2 && parts[0].trim() && parts[1].trim()) {
                            skills[parts[0].trim()] = { '描述': parts[1].trim() };
                        }
                    });
                    return Object.keys(skills).length > 0 ? skills : defaultValue;
                 }
                 if (id === 'items') {
                     const items = {};
                     trimmedValue.split('\n').forEach(line => {
                         if (line.trim()) items[line.trim()] = {};
                     });
                     return Object.keys(items).length > 0 ? items : defaultValue;
                 }
            }
            return trimmedValue;
        };

        const getEquipmentText = (id, defaultValue) => {
            const value = getValue(id, defaultValue);
            return value === '无' ? "无" : value;
        };

        const strength = getValue('strength', 10);
        const agility = getValue('agility', 10);
        const intelligence = getValue('intelligence', 10);
        const endurance = getValue('endurance', 10);
        const spirit = getValue('spirit', 10);
        const charm = getValue('charm', 10);

        const maxHealth = endurance * 10;
        const maxMana = spirit * 5 + intelligence * 5;
        const maxStamina = endurance * 8 + strength * 2;
        const characterData = {
            "护甲值": "请按照装备修改护甲值",
            "物品列表": getValue('items', '无', true),
            "装备栏": {
                "武器": getEquipmentText('weapon', '无'),
                "上身": getEquipmentText('upper-body', '短袖'),
                "下身": getEquipmentText('lower-body', '长裤'),
                "饰品": getEquipmentText('accessory', '无')
            },
            "技能列表": getValue('skills', '无', true),
            "背景描述": getValue('background', '穿越到异世界的中国男大学生', true)
        };

        let outputString = JSON.stringify(characterData, null, 4)
            .replace(/"(\w+)":/g, '"$1": ')
            .replace(/"无"/g, '无')
            .replace(/"物品列表": "无"/, '"物品列表": 无')
            .replace(/"技能列表": "无"/, '"技能列表": 无');

        outputString = outputString.substring(1, outputString.length - 1).trim();
        outputString = outputString.split('\n').map(line => line.substring(4)).join('\n');

        const finalMessage = outputString + '\n\n请按照如上生成角色开局';

        try {
            if (typeof toastr !== 'undefined') toastr.info('稍等，开局生成中');
            const mvudata = Mvu.getMvuData({ type: 'chat' });
            _.set(mvudata,'stat_data.主角.姓名', getValue('name', '{{user}}'));
            _.set(mvudata,'stat_data.主角.种族', getValue('race', '人类'));
            _.set(mvudata,'stat_data.主角.生命值.当前值', maxHealth);
            _.set(mvudata,'stat_data.主角.生命值.最大值', maxHealth);
            _.set(mvudata,'stat_data.主角.魔法值.当前值', maxMana);
            _.set(mvudata,'stat_data.主角.魔法值.最大值', maxMana);
            _.set(mvudata,'stat_data.主角.体力值.当前值', maxStamina);
            _.set(mvudata,'stat_data.主角.体力值.最大值', maxStamina);
            _.set(mvudata,'stat_data.主角.属性列表.力量', strength);
            _.set(mvudata,'stat_data.主角.属性列表.敏捷', agility);
            _.set(mvudata,'stat_data.主角.属性列表.智力', intelligence);
            _.set(mvudata,'stat_data.主角.属性列表.耐力', endurance);
            _.set(mvudata,'stat_data.主角.属性列表.精神', spirit);
            _.set(mvudata,'stat_data.主角.属性列表.魅力', charm);
            await Mvu.replaceMvuData(mvudata  ,{ type: 'message', message_id: 'latest' });
            await setChatMessages([{ message_id: 0, message: "开局初始化完毕" }]);
            await triggerSlash(`/send ${finalMessage}|/trigger`);
            toastr.success('角色开局已生成并发送！！!');

        } catch (error) {
            console.error('函数调用失败:', error);
            if (typeof toastr !== 'undefined') toastr.error('生成失败，请重试。');
            else console.error('生成失败，请检查控制台。'); // Modified for clarity

            console.log("--- Fallback: Generated Message ---");
            console.log(finalMessage);
        }
    });
</script>
