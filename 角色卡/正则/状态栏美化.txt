                        char.生命值 = char.生命值 || {};
                        char.魔法值 = char.魔法值 || {};
                        char.体力值 = char.体力值 || {};
                        char.护甲值 = char.护甲值 || {};
                        char.属性列表 = char.属性列表 || {};
                        char.装备栏 = char.装备栏 || {};
                        char.物品列表 = char.物品列表 || {};
                        char.技能列表 = char.技能列表 || {};
                    }
                    return char;
                }
                catch (e) { console.error('获取主角数据时发生错误:', e); this.error = '无法读取主角数据'; return null; }
            },
            chapterInfo() { return this.chapterInfoData; },
            worldTime() {
                if (!this.mvuData) return {};
                try { return Mvu.getMvuVariable(this.mvuData, '时间', { category: 'stat' }) || {}; } catch(e) { return {}; }
            },
            rewindCount() { return this.mainCharacter?.死亡回溯计数 || 0; },
            // [修改] 强化 computed，过滤掉列表中可能存在的 null/undefined 值
            inventory() {
                if (!this.mainCharacter?.物品列表) return [];
                return Object.values(this.mainCharacter.物品列表).filter(item => item);
            },
            skills() {
                if (!this.mainCharacter?.技能列表) return [];
                return Object.values(this.mainCharacter.技能列表).filter(skill => skill);
            },
            relationships() {
                if (!this.mvuData) return [];
                const rels = Mvu.getMvuVariable(this.mvuData, '关系列表', { category: 'stat' });
                if (!rels) return [];
                return Object.values(rels).filter(char => char && char.姓名); // 确保关系对象有效且有名字
            },
            enemies() {
                if (!this.mvuData) return [];
                const enms = Mvu.getMvuVariable(this.mvuData, '敌人列表', { category: 'stat' });
                if (!enms) return [];
                return Object.values(enms).filter(char => char && char.姓名); // 确保敌人对象有效且有名字
            }
        },
        methods: {
            getPercentage(stat) {
                if (!stat || stat.最大值 == null || stat.最大值 === 0 || stat.当前值 == null) return 0;
                return Math.min(100, Math.max(0, (stat.当前值 / stat.最大值) * 100));
            },
            toggleDetail(id) { this.toggledItems[id] = !this.toggledItems[id]; },
            isDetailVisible(id) { return !!this.toggledItems[id]; },
            updateChapterInfo() {
                try {
                    if (typeof SillyTavern !== 'undefined' && SillyTavern.getCurrentChatId) {
                        const chat_id = SillyTavern.getCurrentChatId();
                        this.chapterInfoData = JSON.parse(window.localStorage.getItem(chat_id) || '{"chapter_name": "未知章节"}');
                    }
                } catch (e) {
                    console.error("无法从 localStorage 更新章节信息:", e);
                    this.chapterInfoData = { chapter_name: '未知章节(错误)' };
                }
            },
            async updateMvuData() {
                if (this.isUpdating) return;
                this.isUpdating = true;
                this.error = null; // [修改] 每次更新前清除旧错误

                try {
                    this.mvuData = await Mvu.getMvuData({ type: 'chat' });
                } catch(e) {
                    this.error = "加载游戏数据失败。请检查控制台获取更多信息。";
                    console.error("MVU Data fetch failed:", e);
                } finally {
                    this.isUpdating = false;
                }
            }
        },
        async mounted() {
            if (typeof Mvu === 'undefined' || typeof eventOn === 'undefined' || typeof SillyTavern === 'undefined') {
                this.error = "环境错误: 缺少 Mvu, eventOn 或 SillyTavern. 请在支持的环境中运行。";
                console.error("Mvu, eventOn, or SillyTavern is not defined.");
                // [修改] 即使环境错误，也设置 mvuData 为一个空对象，避免 v-if="mvuData" 无法渲染错误信息
                this.mvuData = {}; 
                this.updateChapterInfo();
                return;
            }

            this.updateChapterInfo();
            await this.updateMvuData();

            eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, () => {
                this.updateChapterInfo();
                this.updateMvuData();
            });
        }
    });

    app.mount('#app');
    </script>
</body>
</html>
